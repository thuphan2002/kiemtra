git init 			: được sử dụng để tạo một kho lưu trữ trống và mới

git add .			: được dùng để thêm file vào index

git status			: hiển thị danh sách thay đổi của files

git commit			: kết hợp với git add . để cho đánh dấu dự án mới bạn muốn lưu vào repo

git push origin master 	:Để đẩy những thay đổi cục bộ lên nhánh làm việc chính

git clone			: copy một git repository từ remote nguồn

git pull 			: Tải những thay đổi của dự án mới về máy

git remote + link dẫn 		: Tạo một bản sao từ nhánh từ giao tiếp với các remote khác
